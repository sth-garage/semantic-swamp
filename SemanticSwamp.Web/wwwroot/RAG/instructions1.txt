                return String(str)

                    .replace(/</g, "&lt;")
                    .replace(/>/g, "&#62;")
                    .replace(/"/g, "&quot;")
                    .replace(/'/g, "&#39;");
            }

                    : (context.timestamp ?? new Date().toLocaleString());

                const totalTokens = usage?.TotalTokenCount ?? "—";
                const inTokens = usage?.InputTokenCount ?? "—";
                const outTokens = usage?.OutputTokenCount ?? "—";

                const uid = `raw${Date.now()}${Math.floor(Math.random() * 1000)}`;

            /* Append a message to the chat area */
            function addMessage(msgObj, isOwn, latencySeconds) {
                const role = msgObj?.Role?.Label ?? (isOwn ? "Client" : "Assistant");

                let richHtml = '';
                if (msgObj?.Items && msgObj.Items[0]?.Text) {
                    richHtml = msgObj.Items[0].Text;
                } else if (msgObj?.Text) {
                    richHtml = msgObj.Text;
                } else if (typeof msgObj === 'string') {
                    richHtml = msgObj;
                } else {
                    richHtml = $('<div/>').text(JSON.stringify(msgObj)).html();
                }

                const $msgEl = $('<div>')
                    .addClass('chat-message Role-' + role)
                    .attr('data-role', role)
                    .attr('tabindex', 0)
                    .data('chatMessage', msgObj);

                const $bubble = $('<div>').addClass('chat-bubble').html(richHtml);

                if (role !== "Client" && typeof latencySeconds === 'number') {
                    const lbl = `${latencySeconds.toFixed(2)} seconds`;
                    $bubble.append(
                        `<button class="latency-badge" type="button"
                                 data-latency="${latencySeconds}"
                                 aria-label="Response time ${lbl}"
                                 title="View message details">${lbl}</button>`
                    );
                }

                $msgEl.append($bubble);
                $chatArea.append($msgEl);

                // Scroll to bottom for assistant/system messages
                if (role !== "Client") {
                    $chatArea[0].scrollTop = $chatArea[0].scrollHeight;
                }

                if (role === "Assistant") setLatestAnchor($msgEl);

                return $msgEl;
            }

            /* WebSocket connection and message handling */
            function connectSocket() {
                socket = new WebSocket(wsUrl);

                socket.onopen = () => {
                    disableInputs();
                    isWaitingReply = true;
                    try { socket.send('Introduce yourself'); } catch { }
                };

                socket.onclose = () => {
                    disableInputs();
                    addMessage({ Role: { Label: "System" }, Items: [{ Text: "<em>Connection closed.</em>" }] }, false);
                    isWaitingReply = false;
                };

                socket.onerror = () => {
                    disableInputs();
                    addMessage({ Role: { Label: "System" }, Items: [{ Text: "<em>A connection error occurred.</em>" }] }, false);
                    isWaitingReply = false;
                };

                socket.onmessage = event => {
                    isWaitingReply = false;

                    const data = event.data;
                    let obj = null;

                    try { obj = JSON.parse(data); }
                    catch { obj = { Role: { Label: "Assistant" }, Items: [{ Text: data }] }; }

                    const role = obj?.Role?.Label ?? "Assistant";
                    let latencyForThis;

                    if (lastUserMsgTime && role !== "Client") {
                        const now = performance.now();
                        latencyForThis = (now - lastUserMsgTime) / 1000;
                        lastUserMsgTime = null;
                    }

                    addMessage(obj, false, latencyForThis);

                    if (!hasSystemGreeted) {
                        hasSystemGreeted = true;
                        enableInputs(true);
                        $sendMessage.focus();
                    } else {
                        enableInputs(true);
                    }
                };
            }

            /* Send message logic */
            function sendCurrentMessage() {
                if (!socket || socket.readyState !== WebSocket.OPEN || isWaitingReply) return;

                const text = $sendMessage.val();
                if (!text.trim()) return;

                userInputHistory.push(text);
                userInputIndex = userInputHistory.length;

                isWaitingReply = true;
                lastUserMsgTime = performance.now();

                enableInputs(false);

                const $clientMsg = addMessage({
                    Role: { Label: "Client" },
                    Items: [{ Text: $('<div/>').text(text).html() }],
                    AuthorName: "Client"
                }, true);

                // Mark the user's message as anchor (thinking state image)
                setLatestAnchor($clientMsg);

                socket.send(text);

                $sendMessage.val("");
                setTimeout(() => { $sendMessage.val("").focus(); }, 3000);
                $sendButton.prop('disabled', true);
            }

            $('#sendButton').on('click', sendCurrentMessage);

            $('#sendMessage').on('keydown', function (e) {
                if (e.key === "Enter") {
                    e.preventDefault();
                    sendCurrentMessage();
                } else if (e.key === "ArrowUp" || e.key === "ArrowDown") {
                    e.preventDefault();
                    if (e.key === "ArrowUp") {
                        if (userInputIndex > 0) {
                            userInputIndex--;
                            $sendMessage.val(userInputHistory[userInputIndex]);
                        }
                    } else {
                        if (userInputIndex < userInputHistory.length - 1) {
                            userInputIndex++;
                            $sendMessage.val(userInputHistory[userInputIndex]);
                        } else {
                            userInputIndex = userInputHistory.length;
                            $sendMessage.val("");
                        }
                    }
                }
            });

            async function downloadOriginalFile(id) {
                try {
                    const response = await fetch(`/api/DocumentUploads/DownloadOriginalDocument?documentUploadId=${encodeURIComponent(id)}`);
                    if (!response.ok) {
                        alert("Failed to download file!");
                        return;
                    }

                    let downloadFilename = 'download';
                    const disposition = response.headers.get('Content-Disposition') || '';
                    // Attempt to parse filename/filename*
                    const match =
                        /filename\*\s*=\s*UTF-8''([^;]+)|filename\s*=\s*"?([^\";]+)"?/i.exec(disposition);
                    if (match) {
                        downloadFilename = decodeURIComponent((match[1] || match[2] || '').trim());
                    }

                    const blob = await response.blob();
                    const url = window.URL.createObjectURL(blob);

                    const a = document.createElement('a');
                    a.href = url;
                    a.download = downloadFilename || 'download';
                    document.body.appendChild(a);
                    a.click();

                    setTimeout(() => {
                        document.body.removeChild(a);
                        window.URL.revokeObjectURL(url);
                    }, 100);

                } catch (err) {
                    alert("Error: " + err);
                }
            }

            /* Row click expands/collapses (except on download button/icon) */
            $('#documentsTableBody').on('click', 'tr.doc-row', function (e) {
                if ($(e.target).closest('.download-btn').length) return; // ignore download clicks
                toggleDocSummary($(this));
            });

            /* Keyboard support for row expand/collapse */
            $('#documentsTableBody').on('keydown', 'tr.doc-row', function (e) {
                if (e.key === 'Enter' || e.key === ' ') {
                    e.preventDefault();
                    toggleDocSummary($(this));
                }
            });

            /* Download button click handler (stop row toggle) */
            $('#documentsTableBody').on('click', '.download-btn', function (e) {
                e.stopPropagation();
                const id = $(this).closest('.doc-row').attr('data-id');
                if (id) downloadOriginalFile(id);
            });
        });</script>
